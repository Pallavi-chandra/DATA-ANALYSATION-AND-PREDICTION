                      def v(na):
                          rd = pd.read_excel(filename, sheet_name=mod1[na], engine="openpyxl")
                          val1 = np.array(rd)
                          cal1 = np.argwhere(val1 == "Min")
                          row = cal1[0, 0]

                  plt.bar(name2, value2, color='lightblue', width=0.3)
